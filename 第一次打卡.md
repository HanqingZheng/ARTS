func strStr(haystack string, needle string) int {
    hayLen := len(haystack)
    pLen := len(needle)
    if pLen == 0 {return 0}
    if pLen > hayLen {return -1}
    badArr := genBC(needle)
    suffix := make([]int,pLen,pLen)
    prefix := make([]bool,pLen,pLen)
    genGS(needle,pLen,suffix,prefix)
    for i:=0; i <= hayLen - pLen; {
        j := pLen-1;
        for ; j >= 0; j-- {
            if needle[j] != haystack[i+j] {
                break
            }
        }
        if j < 0 {
            return i
        }
        x := j - badArr[haystack[i+j]]
        y := 0
        if j < pLen-1 {
            y = desGS(j,pLen,suffix,prefix)
        }
        i += max([]int{x,y})
    }
    return -1
}
func genBC (s string) []int {
    a := make([]int,256,256)
    for i:=0;i<256;i++ {
        a[i] = -1
    }
    for i:=0;i<len(s);i++ {
        a[int(s[i])] = i 
    }
    return a
}

func genGS (p string, pLen int, suffix []int, prefix []bool) {
    for i:=0; i<pLen-1; i++ {
        j,k := i,0
        for j >= 0 && p[j] == p[pLen-1-k] {
            j--
            k++
            suffix[k] = j+1
        }
        if j <= 0 {
            prefix[k] = true
        }
    }
}

func desGS(index int, pLen int, suffix []int, prefix []bool) int {
    k := pLen-1-index
    if suffix[k] > -1 {
        return index-suffix[k]+1
    }
    for i := index+2; i<pLen; i++ {
        if prefix[pLen-i] {
            return i
        }
    }
    return pLen
}

func max(a []int) int{
    max := a[0]
    for i:=1;i<len(a);i++{
        if a[i]>max{
            max = a[i]
        }
    }
    return max
}
